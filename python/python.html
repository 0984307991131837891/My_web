103數學問題

def is_prime(num):

　　for i in range(2,num//2+1):

　　　if num % i == 0 :

　　　　　return False

　　return True

n = int(input())

for _ in range(n):

　　a,b = list(map(int,input().split(',')))

　　ans = 'N'

　　if b - a >= 2:

　　　　if is_prime(a) and is_prime(b):

　　　　　　ans = "Y"

　　print(ans)

103買郵票

n = int(input())

for i in range(n):

　　a,b,c,d = list(map(int,input().split(',')))

　　for i in range(a+1):

　　　　for j in range(a+1):

　　　　　　if b*i + c*j == d and j+i == a:

　　　　　　　　print(f'{i},{j}')

103在兩字串中,共同出現的英文字母

for i in range(int(input())):

　　Lstr = []

　　string_a = input()

　　string_b = input()

　　if len(string_a) > len(string_b):

　　　　string_a, string_b = string_b, string_a

　　for i in string_a:

　　　　if i in string_b:

　　　　　　Lstr.append(i)

　　Lstr = list(set(Lstr))

　　Lstr.sort()

　　if Lstr != []:

　　　　print(''.join(Lstr))

　　else:

　　　　print('N')
